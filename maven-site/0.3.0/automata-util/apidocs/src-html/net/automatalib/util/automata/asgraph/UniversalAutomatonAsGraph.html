017package net.automatalib.util.automata.asgraph;
018
019import java.util.Collection;
020
021import net.automatalib.automata.UniversalAutomaton;
022import net.automatalib.commons.util.Pair;
023import net.automatalib.graphs.UniversalGraph;
024
025public class UniversalAutomatonAsGraph<S, I, T, SP, TP,A extends UniversalAutomaton<S, I, T, SP, TP>>
026                extends AutomatonAsGraph<S, I, T, A> implements UniversalGraph<S, Pair<I,T>, SP, Pair<I,TP>> {
027        
028        public UniversalAutomatonAsGraph(A automaton,
029                        Collection<? extends I> inputs) {
030                super(automaton, inputs);
031        }
032        
033        @Override
034        public SP getNodeProperty(S node) {
035                return automaton.getStateProperty(node);
036        }
037
038        @Override
039        public Pair<I, TP> getEdgeProperty(Pair<I, T> edge) {
040                return Pair.make(edge.getFirst(),
041                                automaton.getTransitionProperty(edge.getSecond()));
042        }
043
044}
